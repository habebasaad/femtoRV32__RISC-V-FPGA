module SingleMem (input clk, input MemRead, input MemWrite, input [7:0] addr, input [31:0] data_in, [14:12]inst, output reg [31:0] data_out);

reg [7:0] mem [0:255];
integer i;
    initial begin 
        for(i = 0; i < 256; i = i+1)
            mem[i]=0;
    end 
	//address 0->127 instruction memory 
    initial begin 
    //128, 132, 136
    
    // ADDI x0,x0,0
    mem[0] = 8'b00010011; // bits [7:0]
    mem[1] = 8'b00000000; // bits [15:8]
    mem[2] = 8'b00000000; // bits [23:16]
    mem[3] = 8'b00000000; // bits [31:24]
    
    // ADDI x31,x0,128
    mem[4] = 8'b10010011;  // bits [7:0]
    mem[5] = 8'b0_000_1111;    // bits [15:8]
    mem[6] = 8'b00000000; // bits [23:16]
    mem[7] = 8'b00001000; // bits [31:24]
    
    // lw x1, 0(x31)
    mem[8]  = 8'b10000011;  // bits [7:0]
    mem[9]  = 8'b1_010_0000; // bits [15:8]
    mem[10] = 8'b0000_1111; // bits [23:16]
    mem[11] = 8'b00000000; // bits [31:24]
    
    // lw x2, 4(x31)
    mem[12] = 8'b00000011;
    mem[13] = 8'b1_010_0001;
    mem[14] = 8'b0100_1111;
    mem[15] = 8'b00000000;
    
    // not checked 
       // lw x3, 8(x31)
    mem[16] = 8'b1_0000011;
    mem[17] = 8'b1_010_0001;
    mem[18] = 8'b1000_1111;
    mem[19] = 8'b00000000;
    
    // or x4, x1, x2
    mem[20] = 8'b00110011;
    mem[21] = 8'b1_110_0010;
    mem[22] = 8'b0010_0000;
    mem[23] = 8'b00000000;
    
    // beq x4, x3, 4
    mem[24] = 8'b0_1100011;
    mem[25] = 8'b0_000_0100;
    mem[26] = 8'b0011_0010;
    mem[27] = 8'b00000000;
    
    // add x3, x1, x2
    mem[28] = 8'b1_0110011;
    mem[29] = 8'b1_000_0001;
    mem[30] = 8'b0010_0000;
    mem[31] = 8'b00000000;
    
    // add x5, x3, x2
    mem[32] = 8'b1_0110011;
    mem[33] = 8'b1_000_0010;
    mem[34] = 8'b0010_0001;
    mem[35] = 8'b00000000;
    
    // sw x5, 12(x31)
    mem[36] = 8'b0_0100011;
    mem[37] = 8'b1_010_0110;
    mem[38] = 8'b0101_1111;
    mem[39] = 8'b00000000;
    
    // lw x6, 12(x31)
    mem[40] = 8'b0_0000011;
    mem[41] = 8'b1_010_0011;
    mem[42] = 8'b1100_1111;
    mem[43] = 8'b00000000;
    
    // and x7, x6, x1
    mem[44] = 8'b1_0110011;
    mem[45] = 8'b0_111_0011;
    mem[46] = 8'b0001_0011;
    mem[47] = 8'b00000000;
    
    // sub x8, x1, x2
    mem[48] = 8'b0_0110011;
    mem[49] = 8'b1_000_0100;
    mem[50] = 8'b0010_0000;
    mem[51] = 8'b0100000_0;
    
    // add x0, x1, x2
    mem[52] = 8'b0_0110011;
    mem[53] = 8'b1_000_0000;
    mem[54] = 8'b0010_0000;
    mem[55] = 8'b00000000;
    
    // add x9, x0, x1
    mem[56] = 8'b1_0110011;
    mem[57] = 8'b0_000_0100;
    mem[58] = 8'b0001_0000;
    mem[59] = 8'b00000000;
    
    end

	//address 128->255 data memory
    initial begin

        mem[128] = 8'd17; //i
        mem[132] = 8'd9;   // next word (addr 4) n
        mem[136] = 8'd25;  // next word (addr 8) x
    end
    
    always @* begin 
    if ( MemRead) begin 
        case (inst)
        3'b000:   data_out = { {24{mem[addr][7]}}, mem[addr]}; //lb 
        3'b001:   data_out = { {16{mem[addr+1][7]}},mem[addr+1], mem[addr]}; //lh
        3'b010:   data_out = {mem[addr+3],mem[addr+2] ,mem[addr+1], mem[addr]}; //lw
        3'b100:   data_out = { {24'b0}, mem[addr]}; //lbu
        3'b101:   data_out = { {16'b0},mem[addr+1], mem[addr]}; //lhu 
        default:  data_out = 0; 
        endcase
    end 
    end 
    

always @(posedge clk) begin 

    if ( MemWrite)begin
        case (inst) 
        3'b000: begin // SB
                mem[addr] <= data_in[7:0];
            end
        3'b001: begin // SH
                mem[addr]   <= data_in[7:0];
                mem[addr+1] <= data_in[15:8];
            end
        3'b010: begin // SW
                mem[addr]   <= data_in[7:0];
                mem[addr+1] <= data_in[15:8];
                mem[addr+2] <= data_in[23:16];
                mem[addr+3] <= data_in[31:24];
            end
        endcase 
    
    end
end 
endmodule
