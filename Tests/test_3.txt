module SingleMem (input clk, input MemRead, input MemWrite, input [7:0] addr, input [31:0] data_in, [14:12]inst, output reg [31:0] data_out);

reg [7:0] mem [0:255];
integer i;
    initial begin 
        for(i = 0; i < 256; i = i+1)
            mem[i]=0;
    end 
	//address 0->127 instruction memory 
    initial begin 
    //128, 132, 136
        // ADDI x0,x0,0
    mem[0] = 8'b00010011; // bits [7:0]
    mem[1] = 8'b00000000; // bits [15:8]
    mem[2] = 8'b00000000; // bits [23:16]
    mem[3] = 8'b00000000; // bits [31:24]
  
   // ADDI x31,x0,128
    mem[4] = 8'b10010011;  // bits [7:0]
    mem[5] = 8'b0_000_1111;    // bits [15:8]
    mem[6] = 8'b00000000; // bits [23:16]
    mem[7] = 8'b00001000; // bits [31:24]
      
    //mem[1]  = 32'b111111111011_00000_000_00001_0010011; // ADDI x1,x0,-5 -> x1 = -5                                        *ADDI done*
    mem[8]  = 8'b10010011; // 0x93
    mem[9]  = 8'b00000000; // 0x00
    mem[10] = 8'b10110000; // 0xB0
    mem[11] = 8'b11111111; // 0xFF
    
    //mem[2]  = 32'b000000000001_00000_000_00010_0010011; // ADDI x2,x0,1 -> x2 = 1
    mem[12] = 8'b00010011; // 0x13
    mem[13] = 8'b00000001; // 0x01
    mem[14] = 8'b00010000; // 0x10
    mem[15] = 8'b00000000; // 0x00
    
    //mem[3]  = 32'b000000001001_00000_000_00011_0010011; // ADDI x3,x0,9 -> x3 = 9
    mem[16] = 8'b10010011; // 0x93
    mem[17] = 8'b00000001; // 0x03
    mem[18] = 8'b10010000; // 0x90
    mem[19] = 8'b00000000; // 0x00
    
    //mem[4]  = 32'b000000000110_00000_000_00100_0010011; // ADDI x4,x0,6 -> x4 = 6
    mem[20] = 8'b00010011; // 0x13
    mem[21] = 8'b00000010; // 0x02
    mem[22] = 8'b01100000; // 0x60
    mem[23] = 8'b00000000; // 0x00
    
    //mem[4]  = 32'b00001000 0000_0000 0_000_0010 1_0010011; // ADDI x5,x0,128 -> x5 = 128
    mem[24] = 8'b10010011; // 0x13
    mem[25] = 8'b00000010; // 0x05
    mem[26] = 8'b00000000; // 0x80
    mem[27] = 8'b00001000; // 0x00
    
    //mem[5]  = 32'b10000000000000000000_0011 0_0110111; // LUI x6,0x80000  -> x6 = 524288 (decimal) upper 20 to be 2^31      *lui done*
    mem[28] = 8'b00110111; // 0xB7
    mem[29] = 8'b00000011; // 0x06
    mem[30] = 8'b00000000; // 0x00
    mem[31] = 8'b10000000; // 0x80
    
    //mem[6]  = 32'b000000000000_00110_000_0011 1_0010011; // ADDI x7,x6,0 -> x7 = x6 = 2^31
    mem[32] = 8'b10010011; // 0x13
    mem[33] = 8'b00000011; // 0x07
    mem[34] = 8'b00000011; // 0x00
    mem[35] = 8'b00000000; // 0x00
    
    //mem[7] = 32'b000000001011_0000 1_000_1001 1_0010011; // ADDI x19,x1,11 -> x19 = 6
    mem[36] = 8'b10010011; // 0x13
    mem[37] = 8'b10001001; // 0x93
    mem[38] = 8'b10110000; // 0x0B
    mem[39] = 8'b00000000; // 0x00
    
    //mem[8] = 32'b00000000001_0000 1_011_1010 0_0010011; // SLTIU x20,x1,1 -> x20 = 0                                         *SLTIU done*
    mem[40] = 8'b00010011; // 0x13
    mem[41] = 8'b10111010; // 0xA4
    mem[42] = 8'b00010000; // 0x01
    mem[43] = 8'b00000000; // 0x00
    
    //mem[9] = 32'b00000000001_0000 1_010_1010 0_0110011; // SLIT x20,x1,1 -> x20 = 1                                          *SLT done*
    mem[44] = 8'b00010011; // 0x33
    mem[45] = 8'b10101010; // 0xA4
    mem[46] = 8'b00010000; // 0x01
    mem[47] = 8'b00000000; // 0x00
    
    //mem[10] = 32'b0000000 0111_0001 1_100_1010 1_0010011; // XORI x21,x3,7 -> x21 = 14                                         *XORI done* check value for x3
    mem[48] = 8'b10010011; // 0x13
    mem[49] = 8'b11001010; // 0xA9
    mem[50] = 8'b01110001; // 0x07
    mem[51] = 8'b00000000; // 0x00
    
    //mem[11] = 32'b0000000 1000_0001 1_110_1011 0_0010011; // ORI x22,x3,8 -> x22 = 9                                           *ORI done*
    mem[52] = 8'b00010011; // 0x13
    mem[53] = 8'b1_110_1011; // 0xAA
    mem[54] = 8'b1000_0001; // 0x08
    mem[55] = 8'b00000000; // 0x00
    
    //mem[12] = 32'b0000000 0010_0001 1_111_1011 1_0010011; // ANDI x23,x3,2 -> x23 = 0                                          *ANDI done*
    mem[56] = 8'b1_0010011; // 0x13
    mem[57] = 8'b1_111_1011; // 0xAB
    mem[58] = 8'b0010_0001; // 0x02
    mem[59] = 8'b00000000; // 0x00
    
    //mem[13] = 32'b0000000_0 0001_0010 1_001_1100 0_0010011; // SLLI x24,x5,1 -> x24 = 256                                      *SLLI done*
    mem[60] = 8'b00010011; // 0x13
    mem[61] = 8'b1_001_1100; // 0xC4
    mem[62] = 8'b0001_0010; // 0x01
    mem[63] = 8'b00000000; // 0x00
    
    //mem[14] = 32'b0000000_0 0001_0011 0_101_1100 1_0010011; // SRLI x25,x6,1 -> x25 = 2^30 = 1073741824                        *SRLI done*
    mem[64] = 8'b10010011; // 0x13
    mem[65] = 8'b0_101_1100; // 0xC5
    mem[66] = 8'b0001_0011; // 0x01
    mem[67] = 8'b00000000; // 0x00
    
    //mem[15] = 32'b0100000_0 0001_0011 0_101_1101 0_0010011; // SRAI x26,x6,1 -> x26 = -1073741824 (-2^30)                      *SRAI done*
    mem[68] = 8'b00010011; // 0x13
    mem[69] = 8'b0_101_1101; // 0xC6
    mem[70] = 8'b0001_0011; // 0x41
    mem[71] = 8'b0100000_0; // 0x00
    
    //mem[16]  = 32'b0000000_0 0010_0000 1_101_0100 0_1100011; // BGE x1,x2,4 -> not taken                                       *BGE not taken done*
    mem[72] = 8'b01100011; // 0x63
    mem[73] = 8'b1_101_0100; // 0xA4
    mem[74] = 8'b0010_0000; // 0x01
    mem[75] = 8'b00000000; // 0x00
    
    //mem[17] = 32'b0000000_0 0010_0000 1_000_1101 1_0110011; // ADD x27,x1,x2 -> x27 = -4                                       *ADD done*
    mem[76] = 8'b1_0110011; // 0x33
    mem[77] = 8'b1_000_1101; // 0xD8
    mem[78] = 8'b0010_0000; // 0x10
    mem[79] = 8'b00000000; // 0x00
    
    //mem[18] = 32'b0100000_0 0001_0001 0_000_1110 0_0110011; // SUB x28,x2,x1 -> x28 = 6                                        *SUB done*
    mem[80] = 8'b0_0110011; // 0x33
    mem[81] = 8'b0_000_1110; // 0x9C
    mem[82] = 8'b0001_0001; // 0x40
    mem[83] = 8'b01000000; // 0x00
    
    //mem[19] = 32'b0000000_0 0001_0001 0_001_1110 1_0110011; // SLL x29,x2,x1 -> x29 = 134217728 (0x08000000)                   *SLL done*
    mem[84] = 8'b10110011; // 0x33
    mem[85] = 8'b0_001_1110; // 0xF5
    mem[86] = 8'b0001_0001; // 0x02
    mem[87] = 8'b00000000; // 0x00
    
    //mem[20] = 32'b0000000_0 0010_0000 1_010_1111 0_0110011; // SLT x30,x1,x2 -> x30 = 1                                        *SLT done*
    mem[88] = 8'b0_0110011; // 0x33
    mem[89] = 8'b1_010_1111; // 0xFE
    mem[90] = 8'b001000000; // 0x01
    mem[91] = 8'b00000000; // 0x00
    
    //mem[21] = 32'b0000000_0 0010_0000 1_011_1111 1_0110011; // SLTU x31,x1,x2 -> x31 = 0                                       *SLTU done*
    mem[92] = 8'b10110011; // 0x33
    mem[93] = 8'b10111111; // 0xFF
    mem[94] = 8'b00100000; // 0x01
    mem[95] = 8'b00000000; // 0x00
    
    //mem[22] = 32'b0000000_0 0011_0001 0_100_0010 1_0110011; // XOR x5,x2,x3 -> x5 = 8                                          *XOR done*
    mem[96] = 8'b10110011; // 0x33
    mem[97] = 8'b0_100_0010; // 0x29
    mem[98] = 8'b0011_0001; // 0x10
    mem[99] = 8'b00000000; // 0x00
    
    //mem[23]  = 32'b0000000_1 1011_0000 1_110_0100 0_1100011; // BLTU x1,x27,4 -> taken to skip 1 1111011 - 1111100  -4>-5                                *BLTU taken done*
    mem[100] = 8'b01100011; // 0x63
    mem[101] = 8'b11100100; // 0xE4
    mem[102] = 8'b10110000; // 0x01
    mem[103] = 8'b00000001; // 0x00
    
    //mem[24] = 32'b00000000 1011_0000 1_000_1001 1_0010011; // ADDI x19,x1,11 -> x19 = 6 -> skip1
    mem[104] = 8'b1_0010011; // 0x13
    mem[105] = 8'b1_000_1001; // 0x93
    mem[106] = 8'b1011_0000; // 0x0B
    mem[107] = 8'b00000000; // 0x00
    
    // here the value for x7 will be changed 
    //mem[25] = 32'b0000000_00010_0011 0_101_0011 1_0110011; // SRL x7,x6,x2 -> x6 = 1073741824 (0x40000000)                    *SRL done*
    mem[108] = 8'b10110011; // 0x33
    mem[109] = 8'b0_101_0011; // 0xA7
    mem[110] = 8'b0010_0011; // 0x02
    mem[111] = 8'b00000000; // 0x00
    
    //mem[26] = 32'b0100000_1 1100_0011 0_101_0011 1_0110011; // SRA x7,x6,x28 (x28 = 6)-> x7 = -33554432(0xFE000000)            *SRA done*
    mem[112] = 8'b10110011; // 0x33
    mem[113] = 8'b0_101_0011; // 0xA7
    mem[114] = 8'b1100_0011; // 0x40
    mem[115] = 8'b01000001; // 0x00
    
    //mem[27] = 32'b0000000_00010_0000 1_110_0010 0_0110011; // OR x4,x1,x2 -> x4 = -5                                          *OR done*
    mem[116] = 8'b00110011; // 0x33
    mem[117] = 8'b11100010; // 0xC4
    mem[118] = 8'b00100000; // 0x10
    mem[119] = 8'b00000000; // 0x00 
    

    end

	//address 128->255 data memory
    initial begin

        mem[128] = 8'd4; //i
        mem[132] = 8'd5;   // next word (addr 4) n
        mem[136] = 8'd10;  // next word (addr 8) x
    end
    
    always @* begin 
    if ( MemRead) begin 
        case (inst)
        3'b000:   data_out = { {24{mem[addr][7]}}, mem[addr]}; //lb 
        3'b001:   data_out = { {16{mem[addr+1][7]}},mem[addr+1], mem[addr]}; //lh
        3'b010:   data_out = {mem[addr+3],mem[addr+2] ,mem[addr+1], mem[addr]}; //lw
        3'b100:   data_out = { {24'b0}, mem[addr]}; //lbu
        3'b101:   data_out = { {16'b0},mem[addr+1], mem[addr]}; //lhu 
        default:  data_out = 0; 
        endcase
    end 
    end 
    

always @(posedge clk) begin 

    if ( MemWrite)begin
        case (inst) 
        3'b000: begin // SB
                mem[addr] <= data_in[7:0];
            end
        3'b001: begin // SH
                mem[addr]   <= data_in[7:0];
                mem[addr+1] <= data_in[15:8];
            end
        3'b010: begin // SW
                mem[addr]   <= data_in[7:0];
                mem[addr+1] <= data_in[15:8];
                mem[addr+2] <= data_in[23:16];
                mem[addr+3] <= data_in[31:24];
            end
        endcase 
    
    end
end 
endmodule
