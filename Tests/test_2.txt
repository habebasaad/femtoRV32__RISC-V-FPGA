module SingleMem (input clk, input MemRead, input MemWrite, input [7:0] addr, input [31:0] data_in, [14:12]inst, output reg [31:0] data_out);

reg [7:0] mem [0:255];
integer i;
    initial begin 
        for(i = 0; i < 256; i = i+1)
            mem[i]=0;
    end 
	//address 0->127 instruction memory 
    initial begin 
    //128, 132, 136
        // ADDI x0,x0,0
    mem[0] = 8'b00010011; // bits [7:0]
    mem[1] = 8'b00000000; // bits [15:8]
    mem[2] = 8'b00000000; // bits [23:16]
    mem[3] = 8'b00000000; // bits [31:24]
  
   // ADDI x31,x0,128
    mem[4] = 8'b10010011;  // bits [7:0]
    mem[5] = 8'b0_000_1111;    // bits [15:8]
    mem[6] = 8'b00000000; // bits [23:16]
    mem[7] = 8'b00001000; // bits [31:24]
      
       //mem[1]  = 32'b111111111011_00000_000_00001_0010011; // ADDI x1,x0,-5 -> x1 = -5                                        *ADDI done*
    mem[8]  = 8'b10010011; // 0x93
    mem[9]  = 8'b00000000; // 0x00
    mem[10] = 8'b10110000; // 0xB0
    mem[11] = 8'b11111111; // 0xFF
    
    //mem[2]  = 32'b000000000001_00000_000_00010_0010011; // ADDI x2,x0,1 -> x2 = 1
    mem[12] = 8'b00010011; // 0x13
    mem[13] = 8'b00000001; // 0x01
    mem[14] = 8'b00010000; // 0x10
    mem[15] = 8'b00000000; // 0x00
    
    //mem[3]  = 32'b000000001001_00000_000_00011_0010011; // ADDI x3,x0,9 -> x3 = 9
    mem[16] = 8'b10010011; // 0x93
    mem[17] = 8'b00000001; // 0x03
    mem[18] = 8'b10010000; // 0x90
    mem[19] = 8'b00000000; // 0x00
    
    //mem[4]  = 32'b000000000110_00000_000_00100_0010011; // ADDI x4,x0,6 -> x4 = 6
    mem[20] = 8'b00010011; // 0x13
    mem[21] = 8'b00000010; // 0x02
    mem[22] = 8'b01100000; // 0x60
    mem[23] = 8'b00000000; // 0x00
    
    // ADDI x5,x0,128 -> x5 = 128
    mem[24] = 8'b1_0010011;
    mem[25] = 8'b0_000_0010;
    mem[26] = 8'b0000_0000;
    mem[27] = 8'b00001000;
    
    // SB x2,0(x0) -> mem[0] x2 = 1                                    *SB done*
    mem[28] = 8'b0_0100011;
    mem[29] = 8'b1_000_0000;
    mem[30] = 8'b0010_1111;
    mem[31] = 8'b0000000_0;
                          
    //mem[7] = 32'b0000000_00101_00000_000_00001_0100011; // SB x5,1(x0) -> mem[4] x5 =-128 
    mem[32] = 8'b10100011;   // 0x23
    mem[33] = 8'b10000000;  // 0x01
    mem[34] = 8'b01011111;  // 0x40
    mem[35] = 8'b00000000;  // 0x00
                                 
    //mem[8] = 32'b00000000 0000_0000 0_000_0110 1_0000011; // LB x13,0(x0) -> x13 = 1                                          *LB done*
    mem[36] = 8'b10000011; // 0x83
    mem[37] = 8'b10000110; // 0x06
    mem[38] = 8'b00001111; // 0x00
    mem[39] = 8'b00000000; // 0x00
    
    //mem[9] = 32'b000000000000_0000 0_001_0111 0_0000011; // LH x14,0(x0) -> x14 = -32767                                        *LH done*
    mem[40] = 8'b00000011; // 0x03
    mem[41] = 8'b10010111; // 0x17
    mem[42] = 8'b00001111; // 0x00
    mem[43] = 8'b00000000; // 0x00
    
    //mem[10] = 32'b000000000000_0000 0_010_0111 1_0000011; // LW x15,0(x0) -> x15 = 32769                                         *LW done*
    mem[44] = 8'b10000011; // 0x83
    mem[45] = 8'b10100111; // 0x27
    mem[46] = 8'b00001111; // 0x00
    mem[47] = 8'b00000000; // 0x00
    
    //mem[11] = 32'b00000000 0100_0000 0_100_1000 0_0000011; // LBU x16,4(x0) -> x16 = 5 (as written in mem)                                       *LBU done*
    
    mem[48] = 8'b00000011; // 0x03
    mem[49] = 8'b11001000; // 0x48
    mem[50] = 8'b01001111; // 0x40
    mem[51] = 8'b00000000; // 0x00
    
    //mem[12] = 32'b000000000000_0000 0_101_1000 1_0000011; // LHU x17,0(x0) -> x17 = 32769                                         *LHU done*
    mem[52] = 8'b10000011; // 0x03
    mem[53] = 8'b11011000; // 0x51
    mem[54] = 8'b00001111; // 0x00
    mem[55] = 8'b00000000; // 0x00
    
    //mem[13] = 32'b0000000_0 0010_0000 0_000_0100 0_0100011; // SB x2,8(x0) -> mem[8] x2 = 1
    mem[56] = 8'b00100011; // 0x23
    mem[57] = 8'b10000100; // 0x08
    mem[58] = 8'b00101111; // 0x02
    mem[59] = 8'b00000000; // 0x00
    
    //mem[14] = 32'b0000000_00001_0000 0_001_0100 0_0100011; // SH x1,8(x0) -> mem[8,9] x3 =-5  [-5,-1]                                    *SH done*
    mem[60] = 8'b00100011; // 0x23
    mem[61] = 8'b10010100; // 0x08
    mem[62] = 8'b00011111; // 0x10
    mem[63] = 8'b00000000; // 0x00
    
    //mem[15] = 32'b000000000100_0000 0_010_0110 0_0100011; // SW x4,12(x0) -> mem[12] x4 =  6                                  *SW done*
    mem[64] = 8'b00100011; // 0x23
    mem[65] = 8'b10100110; // 0x26
    mem[66] = 8'b01001111; // 0x40
    mem[67] = 8'b00000000; // 0x00
    
    //mem[16] = 32'b000000001100_0000 0_010_0111 1_0000011; // LW x15,12(x0) -> x15 = 6                                         *LW done*
    mem[68] = 8'b10000011; // 0x83
    mem[69] = 8'b10100111; // 0x2F
    mem[70] = 8'b11001111; // 0xC0
    mem[71] = 8'b00000000; // 0x00
    
    end

	//address 128->255 data memory
    initial begin

        mem[128] = 8'd4; //i
        mem[132] = 8'd5;   // next word (addr 4) n
        mem[136] = 8'd10;  // next word (addr 8) x
    end
    
    always @* begin 
    if ( MemRead) begin 
        case (inst)
        3'b000:   data_out = { {24{mem[addr][7]}}, mem[addr]}; //lb 
        3'b001:   data_out = { {16{mem[addr+1][7]}},mem[addr+1], mem[addr]}; //lh
        3'b010:   data_out = {mem[addr+3],mem[addr+2] ,mem[addr+1], mem[addr]}; //lw
        3'b100:   data_out = { {24'b0}, mem[addr]}; //lbu
        3'b101:   data_out = { {16'b0},mem[addr+1], mem[addr]}; //lhu 
        default:  data_out = 0; 
        endcase
    end 
    end 
    

always @(posedge clk) begin 

    if ( MemWrite)begin
        case (inst) 
        3'b000: begin // SB
                mem[addr] <= data_in[7:0];
            end
        3'b001: begin // SH
                mem[addr]   <= data_in[7:0];
                mem[addr+1] <= data_in[15:8];
            end
        3'b010: begin // SW
                mem[addr]   <= data_in[7:0];
                mem[addr+1] <= data_in[15:8];
                mem[addr+2] <= data_in[23:16];
                mem[addr+3] <= data_in[31:24];
            end
        endcase 
    
    end
end 
endmodule
