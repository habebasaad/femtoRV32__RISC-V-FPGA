module SingleMem (input clk, input MemRead, input MemWrite, input [7:0] addr, input [31:0] data_in, [14:12]inst, output reg [31:0] data_out);

reg [7:0] mem [0:255];
integer i;
    initial begin 
        for(i = 0; i < 256; i = i+1)
            mem[i]=0;
    end 
	//address 0->127 instruction memory 
    initial begin 
    //128, 132, 136
        // ADDI x0,x0,0
    mem[0] = 8'b00010011; // bits [7:0]
    mem[1] = 8'b00000000; // bits [15:8]
    mem[2] = 8'b00000000; // bits [23:16]
    mem[3] = 8'b00000000; // bits [31:24]
  
   // ADDI x31,x0,128
    mem[4] = 8'b10010011;  // bits [7:0]
    mem[5] = 8'b0_000_1111;    // bits [15:8]
    mem[6] = 8'b00000000; // bits [23:16]
    mem[7] = 8'b00001000; // bits [31:24]
      
   //mem[0]=32'b000000000000_00000_010_00001_0000011 ; //lw x1, 0(x0) 1
    mem[8]  = 8'b1_0000011; // 0x93
    mem[9]  = 8'b1_010_0000; // 0x00
    mem[10] = 8'b0000_1111; // 0xB0
    mem[11] = 8'b00000000; // 0xFF
    
    //mem[1]=32'b000000000100_00000_010_00010_0000011 ; //lw x2, 4(x0) 5 
    mem[12] = 8'b0_0000011; // 0x13
    mem[13] = 8'b1_010_0001; // 0x01
    mem[14] = 8'b0100_1111; // 0x10
    mem[15] = 8'b00000000; // 0x00
    
    //mem[2]=32'b000000001000_00000_010_00011_0000011 ; //lw x3, 8(x0) 0 
    mem[16] = 8'b1_0000011; // 0x93
    mem[17] = 8'b1_010_0001; // 0x03
    mem[18] = 8'b1000_1111; // 0x90
    mem[19] = 8'b00000000; // 0x00
    
    
    //mem[3]=32'b0000000_00001_00011_000_00011_0110011 ; //add x3,x3,x1 1 > 3> 6 > 10 
    mem[20] = 8'b1_0110011; // 0x13
    mem[21] = 8'b1_000_0001; // 0x04
    mem[22] = 8'b0001_0001; // 0x06
    mem[23] = 8'b00000000; // 0x00
    
    //mem[4]=32'b000000000000_00000_010_00100_0000011; //lw x4, 0(x0) 1 
    mem[24] = 8'b0_0000011; // 0x13
    mem[25] = 8'b1_010_0010; // 0x05
    mem[26] = 8'b00001111; // 0x80
    mem[27] = 8'b00000000; // 0x00
    
    //mem[5]=32'b0000000_00100_00001_000_00001_0110011 ; //add x1, x1, x4 2 >3 > 4> 5  
    mem[28] = 8'b1_0110011; // 0xB7
    mem[29] = 8'b1_000_0000; // 0x06
    mem[30] = 8'b0100_0000; // 0x00
    mem[31] = 8'b00000000; // 0x80
    
    //mem[6]=32'b1_111111_00010_00001_001_1010_1_1100011 ; //bne x1, x2, -6 x 
    mem[32] = 8'b1_1100011; // 0x13
    mem[33] = 8'b1_001_1010_; // 0x07
    mem[34] = 8'b0010_0000; // 0x00
    mem[35] = 8'b1_111111_0; // 0x00
    
    //mem[7]=32'b0000000_00011_00000_010_01000_0100011; //sw x3, 8(x0) x
    mem[36] = 8'b0_0100011; // 0x13
    mem[37] = 8'b1_010_0100; // 0x93
    mem[38] = 8'b0011_1111; // 0x0B
    mem[39] = 8'b00000000; // 0x00
    
    //mem[8]=32'b000000001000_00000_010_00101_0000011; //lw x5, 8(x0) 10 
    mem[40] = 8'b1_0000011; // 0x13
    mem[41] = 8'b1_010_0010; // 0xA4
    mem[42] = 8'b1000_1111; // 0x01
    mem[43] = 8'b00000000; // 0x00
    

    end

	//address 128->255 data memory
    initial begin

        mem[128] = 8'd1; //i
        mem[132] = 8'd5;   // next word (addr 4) n
        mem[136] = 8'd0;  // next word (addr 8) x
    end
    
    always @* begin 
    if ( MemRead) begin 
        case (inst)
        3'b000:   data_out = { {24{mem[addr][7]}}, mem[addr]}; //lb 
        3'b001:   data_out = { {16{mem[addr+1][7]}},mem[addr+1], mem[addr]}; //lh
        3'b010:   data_out = {mem[addr+3],mem[addr+2] ,mem[addr+1], mem[addr]}; //lw
        3'b100:   data_out = { {24'b0}, mem[addr]}; //lbu
        3'b101:   data_out = { {16'b0},mem[addr+1], mem[addr]}; //lhu 
        default:  data_out = 0; 
        endcase
    end 
    end 
    

always @(posedge clk) begin 

    if ( MemWrite)begin
        case (inst) 
        3'b000: begin // SB
                mem[addr] <= data_in[7:0];
            end
        3'b001: begin // SH
                mem[addr]   <= data_in[7:0];
                mem[addr+1] <= data_in[15:8];
            end
        3'b010: begin // SW
                mem[addr]   <= data_in[7:0];
                mem[addr+1] <= data_in[15:8];
                mem[addr+2] <= data_in[23:16];
                mem[addr+3] <= data_in[31:24];
            end
        endcase 
    
    end
end 
endmodule
